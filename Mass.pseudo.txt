CLASS Mass
    PRIVATE
        mass_texture
        mass AS Sprite
        x_position <- 0
        y_position <- 0
        mass_sprite_x_size <- 0
        mass_sprite_y_size <- 0
        vertical_offset <- 0
        loaded <- FALSE
        mass_texture_file_path
        first_load <- TRUE
        is_falling <- TRUE

    PUBLIC
        CONSTRUCTOR Mass
            DECLARE mass_texture_path_components AS a one-dimensional array of 4 elements
            mass_texture_path_components <- ["resources", "images", "anvil.png"]
            mass_texture_file_path <- ARRAY_TO_PATH(mass_texture_path_components)
        END CONSTRUCTOR

        FUNCTION get_game_ceiling
            RETURN y_position + get mass sprite vertical size
        END FUNCTION

        FUNCTION get_x_position
            RETURN x_position
        END FUNCTION

        FUNCTION get_y_position
            RETURN y_position
        END FUNCTION

        FUNCTION get_width
            RETURN get mass sprite horizontal size
        END FUNCTION

        FUNCTION get_height
            RETURN get mass sprite vertical size
        END FUNCTION

        PROCEDURE set_vertical_offset(offset)
            vertical_offset <- offset
        END PROCEDURE

        PROCEDURE compute
            IF loaded is false THEN
                load
            END IF

            DECLARE x_scale
            x_scale <- Registry window_size[1] / mass_sprite_x_size

            set mass scale to (x_scale, x_scale)

            IF mass is falling THEN
                y_position <- -(mass sprite vertical size) + vertical_offset
            END IF

            x_position <- (Registry window_size[0] - (mass sprite horizontal size)) / 2

            IF mass is falling THEN
                vertical_offset <- vertical_offset + 1
            END IF
        END PROCEDURE

        PROCEDURE render(window)
            IF loaded is false THEN
                load
            END IF
            set mass position to (x_position, y_position)
            draw mass sprite to the window
        END PROCEDURE

        PROCEDURE reset
            vertical_offset <- 0
            is_falling <- TRUE
        END PROCEDURE

        PROCEDURE load
            load mass_texture_file_path into mass_texture
            set mass texture to use a smoothing filter

            set the mass sprite texture to mass_texture

            IF this is the first load THEN
                mass_sprite_x_size <- get mass sprite horizontal size
                mass_sprite_y_size <- get mass sprite vertical size
                y_position <- -(mass sprite vertical size)
                x_position <- (Registry window_size[0] - mass_sprite_x_size) / 2
                first_load <- FALSE
            END IF

            loaded <- TRUE
        END PROCEDURE

        PROCEDURE unload
            loaded <- FALSE
            DECLARE new_mass_texture
            set mass sprites texture to new_mass_texture
            set mass texture to new_mass_texture
        END PROCEDURE

        PROCEDURE set_is_falling(is_falling)
            Mass is_falling <- is_falling
        END PROCEDURE

        PROCEDURE set_y_position(y_position)
            Mass y_position <- y_position
        END PROCEDURE

END CLASS
