CLASS LevelTemplate
    PROTECTED
        game_balls AS vector
        cannon_object
        mass_object
        game_lost <- FALSE
        instructions AS LevelInstructions
        left_dragon
        right_dragon
        pop_sounds AS a one-dimensional array of 15 elements
        cannon_fire_sound
        anvil_fail_sound
        level_end_time <- -1
        level_over <- FALSE
        level_complete_text
        score_text

        CONSTRUCTOR LevelTemplate
            DEFINE cannon_fire_sound_path_components AS a one-dimensional array of four elements
            cannon_fire_sound_path_components <- ["resources", "sounds", "cannon fire.ogg"]
            DEFINE anvil_fail_sound_path_components AS a one-dimensional array of four elements
            anvil_fail_sound_path_components <- ["resources", "sounds", "anvil.ogg"]

            FOR i <- 0 TO 15
                DECLARE file
                file <- AS_STRING(i + 1) + ".ogg"
                DECLARE pop_sound_path_components AS a one-dimensional array of four elements
                pop_sound_path_components <- ["resources", "sounds", "pops", file]
                DECLARE file_path
                file_path <- ARRAY_TO_PATH(pop_sound_path_components)
                pop sound at index i to load file_path
            END FOR

            use cannon_fire_sound to load cannon_fire_sound_path_components after converting it to a proper file path
            set the cannon_fire_sound volume to 30%

            use anvil_fail_sound to load anvil_fail_sound_path_components after converting it to a proper file path
            set the anvil_fail_sound volume to 40%
        END CONSTRUCTOR

        PROCEDURE create_ball_grid(game_balls, current_level)
            DECLARE columns
            columns <- 14

            DECLARE rows
            rows <- 6

            IF current_level is Constants LEVEL_TWO THEN
                rows <- 8
            END IF

            DECLARE x_pos
            x_pos <- Registry ball_radius

            DECLARE y_pos
            y_pos <- Registry ball_radius + get game ceiling from mass object

            FOR row <- 0 TO rows
                IF row is an even number THEN
                    columns <- 14
                    x_pos <- Registry ball_radius
                ELSE
                    columns <- 13;
                    x_pos <- Registry ball_radius * 2
                END IF

                FOR column <- 0 TO columns
                    DECLARE game_ball AS Ball WITH x_position <- x_pos + get x_position of mass object, y_position <- y_pos
                    add game_ball to end of game_balls vector
                    x_pos <- x_pos + Registry ball_radius * 2
                END FOR

                y_pos <- y_pos + Registry ball_tesselation_coefficient
            END FOR
        END PROCEDURE

        PROCEDURE anchor_balls_to_mass
            FOR i <- 0 TO get size of game_balls - 2
                IF the ball at index i has y_position sufficiently close to the bottom of the mass object THEN
                    ball at index i is anchored
                END IF
            END FOR

            DEFINE changed
            changed <- TRUE

            WHILE changed DO
                changed <- result of spreading anchor flags across the game_balls vector
            END WHILE

            FOR i <- 0 TO get size of game_balls - 2
                IF game_ball at index i is: not anchored, OR has a velocity in both x AND y of 0, and is not currently falling THEN
                    set ball at index i to fall
                END IF
            END FOR

            FOR i <- 0 TO get size of game_balls - 2
                set game_ball at index i to NOT be anchored
            END FOR
        END PROCEDURE

        PROCEDURE handle_fire_cannon_event(angle)
            IF sound effects are allowed to be played THEN
                play cannon fire sound
            END IF

            SWAP the last element in the vector game_balls, with the second to last element in the vector game_balls

            DECLARE x_velocity
            x_velocity <- SIN(angle) * Constants BALL_SPEED

            DECLARE y_velocity
            y_velocity <- COS(angle) * Constants BALL_SPEED

            set the second to last element in the game_balls vector to have velocity (x_velocity, y_velocity)

            add a new ball to the cannon

            SWAP the last element in the vector game_balls, with the second to last element in the vector game_balls
        END PROCEDURE

    PUBLIC
        FUNCTION run_menu(window, player_input, current_menu)
            DECLARE angle
            angle <- 0

            DECLARE ball_in_motion
            ball_in_motion <- FALSE

            DECLARE index
            index <- 0

            DECLARE random_pop_sound_index
            DECLARE x_pos
            DECLARE y_pos
            DECLARE garbage_ball_elements AS a vector of integers

            compute the instructions
            compute the left dragon
            compute the right dragon

            IF the level is NOT over THEN
                compute the cannon
            END IF

            render the instructions to the window
            render the left dragon to the window
            render the right dragon to the window

            IF the level is NOT over THEN
                render the cannon to the window

                set the score text position on screen to the origin (0, 0)
                RENDER "Score: " + AS_STRING(Registry score) TO window WITH: FONT SIZE 30, COLOUR BLACK

                FOR game_ball IN game_balls
                    IF the index represents the second to last element in the vector game_balls THEN
                        angle <- (get cannon rotation - 90) * Constants DEGREES_TO_RADIANS_CONVERSION_CONSTANT

                        x_pos <- -sin(angle) * Registry ball_radius / 2
                        y_pos <- -cos(angle) * Registry ball_radius / 2
                        set the game_ball to position (get cannon x position + x_pos, get cannon y position + y_pos)
                    END IF

                    compute game_ball
                    render game_ball to window

                    IF the game_ball is popped THEN
                        add index to the end of the vector garbage_ball_elements
                    ELSE IF the game_ball has a velocity of 0, AND is not one of the last two elements in the vector game_balls THEN
                        FOR collision_game_ball IN game_balls
                            check for collisions between collision_game_ball AND game_ball
                        END FOR
                    END IF

                    index <- index + 1
                END FOR

                anchor balls to mass

                index <- 0

                FOR game_ball IN game_balls
                    IF the velocity of the ball is not 0 THEN
                        ball_in_motion <- TRUE
                    END IF

                    IF the game_ball has a velocity of 0, AND is not one of the last two elements in the vector game_balls THEN
                        IF y position for the bottom of the ball is greater than the top of the cannon object THEN
                            IF the ball is NOT falling THEN
                                game_lost <- TRUE
                                level_over <- TRUE
                            END IF
                        END IF
                    END IF

                    IF the pop sound should be played, AND sounds are to be played THEN
                        random_pop_sound_index <- RANDOM INTEGER BETWEEN 0 AND 14
                        play pop sound at index random_pop_sound_index
                    END IF

                    index <- index + 1
                END FOR

                IF the y position for the bottom of the mass, is greater than the y position for the top of the cannon THEN
                    game_lost <- TRUE
                    level_over <- TRUE
                END IF
            END IF

            render mass object to window
            compute mass object

            IF the level is NOT over THEN
                IF the player has made a valid input, AND there are no balls moving currently THEN
                    fire cannon
                END IF
            END IF

            IF all the balls in the grid are popped THEN
                level_over <- TRUE
            END IF

            IF the level_end_time is -1 THEN
                IF game_lost THEN
                    level_end_time <- Registry run_time
                    set the mass object to stop falling
                    set the mass position such that the bottom of the mass is at the bottom of the window

                    IF sounds are to be played THEN
                        play anvil fail sound

                        FOR i <- 0 TO 15
                            random_pop_sound_index <- RANDOM INTEGER BETWEEN 0 AND 14
                            play pop sound at index random_pop_sound_index
                        END FOR
                    END IF
                ELSE IF the level is over THEN
                    set the mass object to stop falling
                    level_end_time <- Registry run time
                END IF
            END IF

            IF level_over AND game NOT lost THEN
                set the level_complete text position on screen to the centre, by specifying location (-1, -1)
                RENDER "Level Complete!" TO window WITH: FONT SIZE 50, COLOUR (60, 176, 67), BOLD
            END IF

            IF the time between the level being either won or lost is greater than three THEN
                IF game lost THEN
                    Registry game_end_state <- Constants LOST
                    RETURN Constants END_MENU
                ELSE IF level over THEN
                    IF the current menu is Constants LEVEL_ONE THEN
                        RETURN Constants LEVEL_TWO
                    ELSE IF the current level is Constants LEVEL_TWO THEN
                        RETURN Constants END_MENU
                    END IF
                END IF
            END IF

            RETURN current_menu
        END FUNCTION

        PROCEDURE unload
            unload cannon
            unload mass
            unload left dragon
            unload right dragon
            unload instructions
            unload score text
            unload level_complete_text
        END PROCEDURE
END CLASS
