START

FUNCTION main
    Set random seed to TIME

    IF read save data is false THEN
        write save data
    END IF

    DECLARE window
    create window with dimensions defined in Registry window size, a caption of "Bouncing Balls!" and disable window resizing
    limit frame rate to 60 hz

    DECLARE menu_navigation AS a one-dimensional array with two elements
    menu_navigation <- [Constants MAIN_MENU, Constants MAIN_MENU]

    DECLARE game_running
    game_running <- TRUE

    DECLARE next_component

    DECLARE main_theme_fade_start
    SET main_theme_fade_start <- -1

    DECLARE window_focus
    SET window_focus <- TRUE

    DECLARE current_volume
    SET current_volume <- 15

    DECLARE player_input

    DECLARE main_menu
    DECLARE level_one
    DECLARE level_two
    DECLARE end_menu
    DECLARE pause_menu

    initialize level_one
    initialize level_two

    DECLARE clouds as a one-dimensional array with 15 elements
    DECLARE ground

    DECLARE main_theme
    DECLARE path_components AS a one-dimensional array with four elements
    path_components <- ["resources", "music", "main theme [extended].ogg"]

    use main_theme to open the file defined in path_components after converting it into a proper file path
    set the main_theme to loop indefinitely
    set the volume of the main_theme to 15%

    IF music is to be played THEN
        play main_theme
    END IF

    WHILE the game is running DO
        DECLARE event
        WHILE there are events to iterate over DO
            IF window closed THEN
                close window
                game_running <- FALSE
                BREAK
            ELSE IF key pressed THEN
                SWITCH key code of key pressed
                    CASE enter key pressed
                        set player_input to register an input from the user
                        BREAK
                    CASE escape key pressed
                        IF current menu is Constants PAUSE_MENU THEN
                            menu_navigation[0] <- menu_navigation[1]
                            menu_navigation[1] <- Constants PAUSE_MENU
                            unload pause menu
                        ELSE IF current menu is Constants LEVEL_ONE OR Constants LEVEL_TWO THEN
                            menu_navigation[1] <- menu_navigation[0]
                            menu_navigation[0] <- Constants PAUSE_MENU
                        ELSE IF current menu is Constants MAIN_MENU THEN
                            menu_navigation[0] <- Constants QUIT
                        ELSE IF current menu is Constants END_MENU THEN
                            menu_navigation[1] <- menu_navigation[0]
                            menu_navigation[0] <- Constants MAIN_MENU
                        END IF
                        BREAK
                END SWITCH
    END WHILE

    IF music should be playing AND isn't currently playing THEN
        play main theme
    ELSE IF music should NOT be playing AND is NOT currently paused THEN
        pause main theme
    END IF

    IF window is currently NOT in focus THEN
        set the volume of main_theme to INTERPOLATE BETWEEN current_volume AND 5 IN DURATION 1 second WITH CURRENT DURATION time since main_theme_fade_start
    ELSE
        set the volume of main_theme to INTERPOLATE BETWEEN current_volume AND 15 IN DURATION 1 second WITH CURRENT DURATION time since main_theme_fade_start
    END IF

    IF window is currently NOT in focus THEN
        IF the current menu is either Constants LEVEL_ONE OR Constants LEVEL_TWO THEN
            menu_navigation[1] <- menu_navigation[0]
            menu_navigation[0] <- Constants PAUSE_MENU
        END IF
    END IF

    set the window background color to Constants SKY_BLUE

    FOR i <- 0 TO 15
        compute cloud at index i
        render cloud at index i to the window
    END FOR

    compute the ground
    render the ground to the window

    IF the current menu is Constants MAIN_MENU THEN
        Registry score <- 0
        next_component <- result from running the main menu
        IF the next_component is NOT Constants MAIN_MENU THEN
            menu_navigation[0] <- next_component
            menu_navigation[1] <- Constants MAIN_MENU
            unload main menu
            reset level one
            reset level two
        END IF
    ELSE IF the current menu is Constants LEVEL_ONE THEN
        next_component <- result from running level one
        IF the next_component is NOT Constants LEVEL_ONE THEN
            menu_navigation[0] <- next_component
            menu_navigation[1] <- Constants LEVEL_ONE
            unload level one
            IF next_component is NOT Constants PAUSE_MENU THEN
                reset level one
            END IF
        END IF
    ELSE IF the current menu is Constants LEVEL_TWO THEN
        next_component <- result from running level two
        IF the next_component is NOT Constants LEVEL_TWO THEN
            menu_navigation[0] <- next_component
            menu_navigation[1] <- Constants LEVEL_TWO
            unload level two
            IF next_component is NOT Constants PAUSE_MENU THEN
                reset level two
            END IF
        END IF
    ELSE IF the current menu is Constants END_MENU THEN
        next_component <- result from running the end menu
        IF the next_component is NOT Constants END_MENU THEN
            menu_navigation[0] <- next_component
            menu_navigation[1] <- Constants END_MENU
            unload end menu
    ELSE IF the current menu is Constants PAUSE_MENU THEN
        next_component <- result from running the pause menu
        IF the next_component is NOT Constants PAUSE_MENU THEN
            menu_navigation[0] <- next_component
            menu_navigation[1] <- Constants PAUSE_MENU
            unload pause menu
            IF next_component is Constants MAIN_MENU THEN
                unload level one
                unload level two
                unload end menu
            END IF
        END IF
    ELSE IF the current menu is Constants QUIT THEN
        game_running <- FALSE
        write save data
    END IF

    display constants drawn to window on screen

    update the player input

    Registry run_time <- Registry run_time + 1 / 60

    RETURN 0
END FUNCTION

END
